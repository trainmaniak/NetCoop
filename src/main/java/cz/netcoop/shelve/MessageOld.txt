package cz.netcoop;

import cz.netcoop.abilities.IAbility;

public class MessageOld {
    public enum Type {
        QUERY,
        REPLY
    }

    private Type type;
    private IAbility ability;
    private String data;

    public IAbility getAbility() {
        return ability;
    }

    public MessageOld(IAbility ability, String data) {
        this.ability = ability;
        this.data = data;
    }

    public static class Builder {
        public static byte[] build(MessageOld messageOld) {
            byte[] result = new byte[256];

            result[0] = (byte) messageOld.type.ordinal();
            result[1] = messageOld.ability.getId();

            messageOld.ability.request();
        }

        public static MessageOld parse(byte[] message) {

        }
    }
}
